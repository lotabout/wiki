The KMP algorithm is used in exact string matching.

  • KMP build a DFA from pattern.
  • KMP simulate DFA with text as inputs.

DFA representation

Use a single state-indexed array next[].

  • Upon character match in state j, go forward to state j+1.
  • Upon character mismatch in state j, go back to state next[j].

Simulation of DFA: 

int j=0;
for (i=0; i<len; i++) {
    if (src[i] == pat[j])
        j++;
    else
        j = next[j];
    if (j == len_of_pat)
        return i-len_of_pat+1; // found
}
return -1;

Modification of DFA representation

Now, the next[j] means if src[i] != pat[j], then we could next compare src[i] with pat[next[j]].

Why 'next' array works?

For example, we have a pattern 'abcd(#)abcd(*)e(j)' and a string "abcdabcda(i)' and now we have str[i] == 'a' != 'e' == pat[j]. And we've already have the pattern 'abcd' matched. So, instead of matching str[1] == 'b' with pat[0] == 'a', we could just matching str[i] == 'a' with pat[4] == 'a'.

How to compute 'next' array?

The essence of computing 'next' array is to find the max string that is both the prefix and postfix of the pattern. Like 'abcd####abcd', 'abcd' is the string to be found. And we have next[12] = 5.

Code to generate 'next' array: 


next[0] = -1;
for (i = 1; i < pat_len; i++) {
    int next_middle = next[i-1];
    while (next_middle >= 0) {
        if (pat[i-1] == pat[next_middle]) {
            break;
        } else {
            next_middle = pat[next_middle];
        }
    }
    next[i] = next_middle + 1;
}

Code to search string: 

i=0;j=0;
while (i<src_len && j<pat_len) {
    if (j==-1 || src[i] == pat[j]) { // move to the next character
        i++;
        j++;
    } else {
        j = next[j];
    }
}

if (j >= pat_len)
    return i-j;
else
    return -1;

An example

#include <stdio.h>
#include <string.h>
#include <malloc.h>


int *build_next(const char *pat, int pat_len)
{
    int i;
    int *next = (int *)malloc(pat_len * sizeof(*next));

    next[0] = -1;
    for (i = 1; i < pat_len; i++) {
        int next_middle = next[i-1];
        while (next_middle >= 0) {
            if (pat[i-1] == pat[next_middle]) {
                break;
            } else {
                next_middle = pat[next_middle];
            }
        }
        next[i] = next_middle + 1;
    }

    return next;
}

int KMP_Search(const char *src, const char *pat, const int *next)
{
    int src_len = strlen(src);
    int pat_len = strlen(pat);

    int i,j;

    i=0;
    j=0;
    while (i<src_len && j<pat_len) {
        if (j==-1 || src[i] == pat[j]) {
            i++;
            j++;
        } else {
            j = next[j];
        }
    }
    if (j >= pat_len) {
        return i - pat_len;
    } 

    return -1;
}

int KMP(const char *src, const char *pat)
{
    int *next = build_next(pat, strlen(pat));
    int ret = KMP_Search(src, pat, next);
    free(next);

    return ret;
}

int main(int argc, const char *argv[])
{
    
    const char *src = "aaaaaaaaaaaaaaacaaaaaaaaaaaaaaab";
    const char *pat = "aaaab";

    printf("%d\n", KMP(src, pat));
    return 0;
}
